* Rôle de la Vue
Fournit une visualisation graphique de l'évolution des finances (Revenus cumulés, Dépenses cumulées et Solde) au fil du temps.

* Préparation des Données
Le calcul est fait en cumulant les transactions jour après jour. On utilise =groupby= de Pandas pour agréger les montants si plusieurs transactions ont eu lieu le même jour.

#+begin_src python :tangle ../views/charts.py
import flet as ft
import flet_charts as fc
import pandas as pd
import database

def get_charts_view():
    df = database.get_df()
    if df.empty:
        return ft.Center(content=ft.Text("Tsy mbola misy angona."))

    df['Date'] = pd.to_datetime(df['Date'])
    df['Type'] = df['Type'].replace({'Income': 'Miditra', 'Expense': 'Fandaniana'})
    
    # Agrégation journalière
    daily_df = df.groupby(['Date', 'Type'])['Amount'].sum().unstack(fill_value=0)
    for col in ['Miditra', 'Fandaniana']:
        if col not in daily_df.columns: daily_df[col] = 0
            
    daily_df = daily_df.sort_index()
#+end_src
